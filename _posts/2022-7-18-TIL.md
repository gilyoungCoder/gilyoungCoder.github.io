## 카카오 인턴쉽 코테



```javascript
function solution(board, moves) {
    var n = board.length;
    var stack=[];
    var res=[];
    var answer = 0;
    
    for (var i = 0; i <n; i++) {
        stack.push([]);
    }
    
    for(let i=0; i<n; i++){
        for(let j=n-1; j>=0; j--){
            if(board[j][i]==0){
                break;
            }
            stack[i].push(board[j][i])
        }
    }
    
    for(k of moves){
        if(stack[k-1].length==0){
            continue;
        }
        let temp = stack[k-1].pop();
        if(res.length>0&&res[res.length-1]==temp){
            answer++;
            res.pop();
        }else{
        res.push(temp);
        }
    }
    
    return answer*2;
}
```
